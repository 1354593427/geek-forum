一、背景与概述

Vision-Language-Action (VLA) 模型正在重塑机器人学习领域。从 OpenVLA 到 RT-2，这些模型通过融合视觉感知、语言理解和动作预测，实现了前所未有的泛化能力。

然而，VLA 模型的训练需要大量高质量的机器人操作数据。MuJoCo 作为物理仿真领域的标杆工具，为我们提供了低成本、高效率的数据采集方案。本文将详细介绍如何将 MuJoCo 仿真数据转换为 VLA 模型可微调的 RLDS 格式数据集。

二、VLA 数据集格式要求

当前主流的 VLA 模型（如 OpenVLA、Octo）普遍采用 RLDS (Reinforcement Learning Datasets) 格式，这是 Google 推出的基于 TensorFlow Datasets 的标准格式。

2.1 RLDS 数据格式核心要素

一个标准的 RLDS episode 包含以下字段：

{
    'steps': [
        {
            'observation': {
                'image': uint8_array,           # RGB 图像 (H, W, 3)
                'depth': float32_array,         # 深度图 (可选)
                'state': float32_array,         # 机器人状态 (关节角度等)
            },
            'action': float32_array,            # 动作向量
            'reward': scalar,                   # 奖励值
            'is_terminal': bool,                # 是否终止
            'is_first': bool,                   # 是否首帧
            'is_last': bool,                    # 是否末帧
            'language_instruction': string,     # 语言指令
        },
        # ... 更多步骤
    ]
}

2.2 Open X-Embodiment 数据集规范

Open X-Embodiment 是目前最大的开源机器人数据集联盟，其数据格式已成为事实标准。关键特征包括：

  • 统一的动作空间：绝对/相对位置控制，维度通常为 7-8 (x, y, z, roll, pitch, yaw, gripper)
  • 图像分辨率：224x224 或 256x256 是 VLA 模型的标准输入
  • 语言标注：每个 episode 需要对应自然语言指令

三、MuJoCo 数据采集实现

3.1 环境配置与基础采集

以下是一个完整的 MuJoCo 数据采集示例，使用 mujoco 和 dm_control：

import mujoco
import numpy as np
from dm_control import suite
from PIL import Image
import h5py
from collections import defaultdict

class MuJoCoDataCollector:
    """MuJoCo 数据采集器 - 用于 VLA 训练数据准备"""
    
    def __init__(self, domain_name='manipulator', task_name='reach'):
        self.env = suite.load(domain_name, task_name)
        self.physics = self.env.physics
        
    def collect_episode(self, policy_func, language_instruction, max_steps=1000):
        """
        采集单个 episode 数据
        
        Args:
            policy_func: 策略函数，输入观测，输出动作
            language_instruction: 语言指令字符串
            max_steps: 最大步数
        """
        episode_data = defaultdict(list)
        time_step = self.env.reset()
        step_count = 0
        
        while not time_step.last() and step_count < max_steps:
            # 获取观测
            obs = self._get_observation(time_step)
            
            # 执行策略获取动作
            action = policy_func(obs)
            
            # 记录数据
            episode_data['observations'].append(obs)
            episode_data['actions'].append(action)
            episode_data['rewards'].append(time_step.reward)
            episode_data['language_instructions'].append(language_instruction)
            
            # 环境步进
            time_step = self.env.step(action)
            step_count += 1
        
        return episode_data
    
    def _get_observation(self, time_step):
        """提取观测数据"""
        obs = {}
        
        # RGB 图像渲染 (224x224 用于 VLA 模型)
        obs['image'] = self.physics.render(height=224, width=224, camera_id=0)
        
        # 深度图 (可选)
        obs['depth'] = self.physics.render(height=224, width=224, camera_id=0, depth=True)
        
        # 机器人状态 (关节角度、末端执行器位置等)
        obs['state'] = np.concatenate([
            self.physics.data.qpos,      # 关节位置
            self.physics.data.qvel,      # 关节速度
        ])
        
        return obs

3.2 多视角图像采集

VLA 模型通常受益于多视角输入。以下是配置多个相机的示例：

def setup_cameras(self):
    """配置多相机视角"""
    self.camera_configs = [
        {'name': 'front', 'camera_id': 0, 'azimuth': 180, 'elevation': -30},
        {'name': 'top', 'camera_id': 1, 'azimuth': 0, 'elevation': -90},
        {'name': 'side', 'camera_id': 2, 'azimuth': 90, 'elevation': -30},
    ]

def render_multiview(self):
    """渲染多视角图像"""
    images = {}
    for config in self.camera_configs:
        img = self.physics.render(
            height=224, 
            width=224, 
            camera_id=config['camera_id']
        )
        images[config['name']] = img
    return images

4.2 创建 RLDS Dataset Builder

import tensorflow_datasets as tfds
import rlds
from typing import Dict, Any

class MuJoCoVLAConfig(tfds.core.BuilderConfig):
    """MuJoCo VLA 数据集配置"""
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.action_dim = 7  # (x, y, z, roll, pitch, yaw, gripper)
        self.state_dim = 14  # 根据具体机器人调整

class MuJoCoVLADataset(tfds.core.GeneratorBasedBuilder):
    """
    MuJoCo 采集数据转换为 RLDS 格式
    用于 OpenVLA / Octo 等模型微调
    """
    VERSION = tfds.core.Version('1.0.0')
    
    BUILDER_CONFIGS = [
        MuJoCoVLAConfig(
            name='reach_task',
            description='MuJoCo reach task with language instructions',
        ),
        MuJoCoVLAConfig(
            name='pick_place',
            description='MuJoCo pick and place task',
        ),
    ]
    
    def _info(self) -> tfds.core.DatasetInfo:
        """定义数据集结构"""
        return self.dataset_info_from_configs(
            features=tfds.features.FeaturesDict({
                'steps': tfds.features.Dataset({
                    'observation': tfds.features.FeaturesDict({
                        'image': tfds.features.Image(
                            shape=(224, 224, 3),
                            dtype=np.uint8,
                            encoding_format='png',
                        ),
                        'state': tfds.features.Tensor(
                            shape=(self.builder_config.state_dim,),
                            dtype=np.float32,
                        ),
                    }),
                    'action': tfds.features.Tensor(
                        shape=(self.builder_config.action_dim,),
                        dtype=np.float32,
                    ),
                    'reward': np.float64,
                    'is_terminal': np.bool_,
                    'is_first': np.bool_,
                    'is_last': np.bool_,
                    'language_instruction': tfds.features.Text(),
                }),
            }),
            supervised_keys=None,
        )
    
    def _split_generators(self, dl_manager):
        """定义数据分割"""
        # 加载已采集的 MuJoCo 数据
        data_path = dl_manager.manual_dir / 'mujoco_collected_data.hdf5'
        
        return {
            'train': self._generate_examples(data_path, split='train'),
            'val': self._generate_examples(data_path, split='val'),
        }
    
    def _generate_examples(self, data_path, split):
        """生成 episode 数据"""
        import h5py
        
        with h5py.File(data_path, 'r') as f:
            episodes = f[split]
            
            for episode_id, episode_data in episodes.items():
                steps = []
                
                for i in range(len(episode_data['actions'])):
                    step = {
                        'observation': {
                            'image': episode_data['images'][i],
                            'state': episode_data['states'][i],
                        },
                        'action': episode_data['actions'][i],
                        'reward': episode_data['rewards'][i],
                        'is_terminal': i == len(episode_data['actions']) - 1,
                        'is_first': i == 0,
                        'is_last': i == len(episode_data['actions']) - 1,
                        'language_instruction': episode_data['language_instruction'][()].decode(),
                    }
                    steps.append(step)
                
                yield episode_id, {'steps': steps}

4.3 数据转换脚本

#!/usr/bin/env python3
"""
MuJoCo HDF5 数据转换为 RLDS 格式
"""
import os
import numpy as np
import h5py
import tensorflow_datasets as tfds
from pathlib import Path

def convert_to_rlds(input_hdf5: str, output_dir: str, dataset_name: str = 'mujoco_vla'):
    """
    将 MuJoCo 采集的 HDF5 数据转换为 RLDS 格式
    
    Args:
        input_hdf5: 输入 HDF5 文件路径
        output_dir: 输出 RLDS 数据集目录
        dataset_name: 数据集名称
    """
    output_path = Path(output_dir) / dataset_name
    output_path.mkdir(parents=True, exist_ok=True)
    
    # 设置手动下载目录
    os.environ['TFDS_DATA_DIR'] = str(output_path)
    
    # 构建数据集
    builder = tfds.builder(
        'mujoco_vla_dataset',
        data_dir=str(output_path)
    )
    
    # 下载并准备数据
    builder.download_and_prepare(
        download_config=tfds.download.DownloadConfig(
            manual_dir=Path(input_hdf5).parent
        )
    )
    
    print(f"RLDS dataset saved to: {output_path}")
    
    # 验证数据集
    ds = builder.as_dataset(split='train')
    example = next(iter(ds))
    print(f"Sample episode steps: {len(example['steps'])}")
    print(f"Action shape: {example['steps'][0]['action'].shape}")
    print(f"Image shape: {example['steps'][0]['observation']['image'].shape}")

if __name__ == '__main__':

        input_hdf5='./mujoco_data/collected_data.hdf5',

六、最佳实践与注意事项

6.1 Sim-to-Real 迁移建议

  • 域随机化：在 MuJoCo 中随机化纹理、光照、相机位置，增强泛化能力
  • 动作空间对齐：确保仿真动作与真实机器人控制方式一致（绝对/相对位置控制）
  • 延迟建模：在仿真中引入控制延迟，模拟真实系统特性

6.2 数据质量检查清单

  • 图像分辨率统一为 224x224 或 256x256
  • 动作值归一化到 [-1, 1] 范围
  • 语言指令多样化，避免单一模板
  • 每个 episode 长度适中（50-500 步）
  • 验证集与训练集场景不重叠
